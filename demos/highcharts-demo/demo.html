图表点击事件，在plotOptions添加click事件在points.events.click下定义时，可直接用this引用点击点；直接用events.click定义，则需用event.point引用点击点。

series.data中添加自定义参数：

series: [{
	type: 'pie',
	name: 'Browser share',
	data: [
	    {
	        name: 'Firefox',
	        y: 45.0,
	        cn_name: '火狐'
	    }]
}];

plotOptions添加点击事件：

plotOptions: {
    pie: {
        allowPointSelect: true,
        cursor: 'pointer',
        depth: 35,
        dataLabels: {
            enabled: true,
            format: '{point.name}'
        },
        events: {
        	click: function(event) {
        		var desc = "扇区信息
名称：" + event.point.name + "
中文名：" + event.point.cn_name + "
值：" + event.point.y + "
比例：" + event.point.percentage + " %";
        		addNotify("gritter", "点击事件通知", desc, '5000', null);
        	}
        }
    }
},
